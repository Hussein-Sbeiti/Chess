    ("knight", "white"): 2.0,
    ("bishop", "white"): 3.0,
    ("rook", "white"): 4.0,
    ("queen", "white"): 5.0,
    ("king", "white"): 6.0,
    ("pawn", "black"): -1.0,
    ("knight", "black"): -2.0,
    ("bishop", "black"): -3.0,
    ("rook", "black"): -4.0,
    ("queen", "black"): -5.0,
    ("king", "black"): -6.0,
}
def encode_board_only(state) -> np.ndarray:
    vec = []
    for row in state.board:
        for piece in row:
            if piece is None:
                vec.append(0.0)
            else:
                vec.append(PIECE_TO_VALUE.get((piece.kind, piece.color), 0.0))
    return np.array(vec, dtype=np.float32)
def encode_state(state) -> np.ndarray:
    board_vec = encode_board_only(state)
    side_to_move = np.array([1.0 if state.turn == "white" else -1.0], dtype=np.float32)
    castling = np.array([
        1.0 if getattr(state, "white_can_castle_kingside", False) else 0.0,
        1.0 if getattr(state, "white_can_castle_queenside", False) else 0.0,
        1.0 if getattr(state, "black_can_castle_kingside", False) else 0.0,
        1.0 if getattr(state, "black_can_castle_queenside", False) else 0.0,
    ], dtype=np.float32)
    en_passant = np.array([
        1.0 if getattr(state, "en_passant_target", None) is not None else 0.0
    ], dtype=np.float32)
    return np.concatenate([board_vec, side_to_move, castling, en_passant])

import numpy as np
class TinyChessNet:
    def __init__(self, input_size=70, hidden1=64, hidden2=32, seed=42):
        rng = np.random.default_rng(seed)
        self.W1 = rng.normal(0, 0.05, (input_size, hidden1)).astype(np.float32)
        self.b1 = np.zeros((1, hidden1), dtype=np.float32)
        self.W2 = rng.normal(0, 0.05, (hidden1, hidden2)).astype(np.float32)
        self.b2 = np.zeros((1, hidden2), dtype=np.float32)
        self.W3 = rng.normal(0, 0.05, (hidden2, 1)).astype(np.float32)
        self.b3 = np.zeros((1, 1), dtype=np.float32)
    def relu(self, x):
        return np.maximum(0, x)
    def relu_derivative(self, x):
        return (x > 0).astype(np.float32)
    def tanh(self, x):
        return np.tanh(x)
    def forward(self, x):
        if x.ndim == 1:
            x = x.reshape(1, -1)
        z1 = x @ self.W1 + self.b1
        a1 = self.relu(z1)
        z2 = a1 @ self.W2 + self.b2
        a2 = self.relu(z2)
        z3 = a2 @ self.W3 + self.b3
        y = self.tanh(z3)
        cache = (x, z1, a1, z2, a2, z3, y)
        return y, cache
    def predict(self, x):
        y, _ = self.forward(x)
        return float(y[0, 0])
    def train_step(self, x, target, lr=0.001):
        if x.ndim == 1:
            x = x.reshape(1, -1)
        target = np.array([[target]], dtype=np.float32)
        y, cache = self.forward(x)
        x, z1, a1, z2, a2, z3, _ = cache
        dy = 2.0 * (y - target)
        dz3 = dy * (1.0 - np.tanh(z3) ** 2)
        dW3 = a2.T @ dz3
        db3 = dz3
        da2 = dz3 @ self.W3.T
        dz2 = da2 * self.relu_derivative(z2)
        dW2 = a1.T @ dz2
        db2 = dz2
        da1 = dz2 @ self.W2.T
        dz1 = da1 * self.relu_derivative(z1)
        dW1 = x.T @ dz1
        db1 = dz1
        self.W3 -= lr * dW3
        self.b3 -= lr * db3
        self.W2 -= lr * dW2
        self.b2 -= lr * db2
        self.W1 -= lr * dW1
        self.b1 -= lr * db1
        loss = float(np.mean((y - target) ** 2))
        return loss
    def save(self, path):
        np.savez(
            path,
            W1=self.W1, b1=self.b1,
            W2=self.W2, b2=self.b2,
            W3=self.W3, b3=self.b3,
        )
    def load(self, path):
        data = np.load(path)
        self.W1 = data["W1"]
        self.b1 = data["b1"]
        self.W2 = data["W2"]
        self.b2 = data["b2"]
        self.W3 = data["W3"]
        self.b3 = data["b3"]
